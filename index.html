    <script>
        // 1. Анимация появления и выхода из размытия (Blur Reveal)
        document.addEventListener("DOMContentLoaded", () => {
            const observerOptions = {
                root: null,
                rootMargin: '0px',
                threshold: 0.15 // Элемент срабатывает, когда 15% его площади видно на экране
            };

            const observer = new IntersectionObserver((entries, observer) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        entry.target.classList.add('active');
                        // Если нужно, чтобы элементы снова размывались при скролле вверх, 
                        // уберите строку ниже. Но для Landing Page лучше оставить так:
                        observer.unobserve(entry.target); 
                    }
                });
            }, observerOptions);

            const revealElements = document.querySelectorAll('.blur-reveal');
            revealElements.forEach(el => observer.observe(el));
        });

        // 2. Функция копирования адреса в буфер обмена (Микро-интеракция)
        function copyText(elementId, btnElement) {
            const textToCopy = document.getElementById(elementId).innerText;
            
            navigator.clipboard.writeText(textToCopy).then(() => {
                const originalText = btnElement.innerText;
                btnElement.innerText = "Скопировано!";
                btnElement.style.background = "#34c759";
                btnElement.style.color = "#fff";
                btnElement.style.border = "1px solid #34c759";
                
                setTimeout(() => {
                    btnElement.innerText = originalText;
                    btnElement.style.background = "#fff";
                    btnElement.style.color = "inherit";
                    btnElement.style.border = "1px solid rgba(0,0,0,0.1)";
                }, 2000);
            }).catch(err => {
                console.error('Ошибка копирования: ', err);
            });
        }
    </script>
